```python
    total_questions: int,
    topics_data: Dict[int, int]  # topic_id -> available_final_questions
) -> Dict[int, int]:
    """
    –†–∞—Å–ø—Ä–µ–¥–µ–ª—è–µ—Ç –≤–æ–ø—Ä–æ—Å—ã –º–µ–∂–¥—É —Ç–µ–º–∞–º–∏ —Å –∞–¥–∞–ø—Ç–∏–≤–Ω—ã–º –æ–∫—Ä—É–≥–ª–µ–Ω–∏–µ–º.

    –ü—Ä–∏–º–µ—Ä: 10 –≤–æ–ø—Ä–æ—Å–æ–≤, 3 —Ç–µ–º—ã (15, 8, 4 –≤–æ–ø—Ä–æ—Å–∞ –¥–æ—Å—Ç—É–ø–Ω–æ)
    –†–µ–∑—É–ª—å—Ç–∞—Ç: {topic1: 5, topic2: 3, topic3: 2}
    """

    total_available = sum(topics_data.values())
    if total_available < total_questions:
        # –ù–µ–¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ –≤–æ–ø—Ä–æ—Å–æ–≤ - —Ä–∞—Å–ø—Ä–µ–¥–µ–ª—è–µ–º –ø—Ä–æ–ø–æ—Ä—Ü–∏–æ–Ω–∞–ª—å–Ω–æ
        return topics_data

    distribution = {}
    remaining = total_questions

    for topic_id, available in topics_data.items():
        if remaining <= 0:
            distribution[topic_id] = 0
            continue

        # –ü—Ä–æ–ø–æ—Ä—Ü–∏–æ–Ω–∞–ª—å–Ω–æ–µ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ —Å –æ–∫—Ä—É–≥–ª–µ–Ω–∏–µ–º

        allocated = min(remaining, max(1, round(total_questions * proportion)))
        distribution[topic_id] = allocated
        remaining -= allocated

    return distribution
```
